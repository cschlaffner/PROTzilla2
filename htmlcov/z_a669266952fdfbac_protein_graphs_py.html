376def _longest_paths(protein_graph: nx.DiGraph, start_node: str): 

377 """ 

378 Create a mapping from node to longest_path from source to that node. 

379 

380 Let n be a node in the graph and P be the set of all predecessors of n.

382

384 __start__ -> ABC -> DE -> F -> JK -> __end__

614 

615 if not found_lines: 

616 raise ValueError(f"Could not find lines with Sequence in {protein_path}") 

617 

618 reference_sequence = "" 

619 seq_len = None 

620 for line in found_lines:

622 # part of the sequence 

623 if line.startswith("SQ"): 

624 line = line.split() 624 ↛ 625line 624 didn't jump to line 625, because the condition on line 624 was never true

625 seq_len = int(line[2]) 

626 continue 

627 if line.startswith("//"): 

628 break 

629 reference_sequence += line.strip() 

630 reference_sequence = reference_sequence.replace(" ", "") 

631 631 ↛ 632line 631 didn't jump to line 632, because the condition on line 631 was never true

634 if seq_len is None or not isinstance(seq_len, int) or seq_len < 1: 

635 raise ValueError( 

636 f"Could not find sequence length for protein file at path {protein_path}" 

637 )

642def _potential_peptide_matches( 

643 allowed_mismatches: int, k: int, peptides: list, ref_index: dict, seq_len: int 

644): 

645 """ 

646 Get potential start positions for peptides on reference sequence. This is done by 

647 looking up the k-mers of the peptide in the reference sequence index. 

648 

649 :param allowed_mismatches: number of mismatches allowed per peptide-match try 

650 :type allowed_mismatches: int 

651 :param k: size of kmer 

652 :type k: int 

653 :param peptides: list of peptide-strings 

654 :type peptides: list 

655 :param ref_index: mapping from kmer to match-positions on reference sequence 

656 :type ref_index: dict(kmer: [starting position]} 

657 

658 :return: dict(peptide: [match start on reference sequence]), 

659 list(peptides without match) 

660 :rtype: dict, list 

661 """ 

662 

663 if not isinstance(k, int) or k < 1: 

664 raise ValueError(f"k must be positive integer, but is {k}") 

665 if not isinstance(allowed_mismatches, int) or allowed_mismatches < 0: 

666 raise ValueError( 

667 f"allowed_mismatches must be non-negative integer, " 

668 f"but is {allowed_mismatches}" 

669 ) 

670 

671 logger.debug("Matching peptides to reference sequence") 

672 potential_peptide_matches = {} 

673 peptide_mismatches = set() 

674 for peptide in peptides: 

675 kmer = peptide[:k] 

676 matched_starts = [] 

677 if kmer not in ref_index: 

678 peptide_mismatches.add(peptide) 

679 continue 

680 for match_start_pos in ref_index[kmer]: 

681 if match_start_pos + len(peptide) > seq_len: 

682 # for now potential matches like this will be dismissed even if 

683 # match_start_pos + len(peptide) - allowed_mismatches <= seq_len 

684 logger.debug( 

685 f"match would be out of bounds for peptide {peptide}, " 

686 f"match_start_pos {match_start_pos}" 

687 ) 

688 continue 

689 matched_starts.append(match_start_pos) 

690 

691 if matched_starts: 

692 potential_peptide_matches[peptide] = matched_starts 

693 else: 

694 peptide_mismatches.add(peptide) 

695 

696 logger.debug( 

697 f"potential peptide matches - peptide:[starting_pos] :: " 

698 f"{potential_peptide_matches}" 

699 ) 

700 logger.debug(f"peptide mismatches: {peptide_mismatches}") 

701 

702 return potential_peptide_matches, sorted(list(peptide_mismatches))

705def _create_contigs_dict(node_start_end: dict): 

706 """ 

707 Creates mapping from node to start and end position of contigs as well as the 

708 peptide(s) that is responsible for the match. 

709 

710 :param node_start_end: dict of peptide to dict of start index of peptide match to 

711 dict of node to tuple of start and end positions of matches within the node 

712 :type node_start_end: dict[str, dict[int, dict[str, tuple[int, int]]]] 

713 

714 :return: dict of node to list of triple of start position, end position and 

715 peptide(s) responsible for match 

716 """ 

717 

718 node_match_data = defaultdict(lambda: {"match_locations": []}) 

719 for peptide, start_pos_dict in node_start_end.items(): 

720 for start_index, node_dict in start_pos_dict.items(): 

721 for node, start_end in node_dict.items(): 

722 node_match_data[node]["match_locations"].append( 

723 (start_end[0], start_end[1], peptide) 

724 ) 

725 

726 for node in node_match_data: 

727 node_match_data[node]["match_locations"].sort() 

728 

729 # merge tuples where start of second tuple is smaller or equal to end of first tuple 

730 node_mod = {} 

731 for node, node_dict in node_match_data.items(): 

732 new_positions_list = [] 

733 for start, end, peptide in node_dict["match_locations"]: 

734 if not new_positions_list: 

735 new_positions_list.append((start, end, peptide)) 

736 elif start <= new_positions_list[-1][1]: 

737 new_positions_list[-1] = ( 

738 new_positions_list[-1][0], 

739 end, 

740 new_positions_list[-1][2] + ";" + peptide, 

741 ) 

742 else: 

743 new_positions_list.append((start, end, peptide)) 

744 

745 node_mod[node] = new_positions_list 

746 

747 return node_mod

750def _match_potential_matches( 

751 potential_peptide_matches, 

752 graph_index, 

753 peptide_mismatches, 

754 allowed_mismatches, 

755 graph, 

756 longest_paths, 

757): 

758 """ 

759 Matches the potential peptide matches to the graph. This function utilizes a 

760 recursive matching method that branches off at the end of each node that is matches 

761 to the end while there is still amino acids left over in the peptide. A new 

762 recursion is opened for each successor of a node, given the scenario above. 

763 A recursion is closed if the end of the peptide is reached or if the number of 

764 mismatches exceeds the allowed number of mismatches. This leads to potential run 

765 time problems for many allowed mismatches and long peptides in graphs with frequent 

766 Variations. 

767 

768 For the recursive method, see below. 

769 

770 :param potential_peptide_matches: dict of peptide to list of starting positions 

771 :type potential_peptide_matches: dict[str, list[int]] 

772 :param graph_index: list of lists, each list contains the nodes and AAs at that 

773 given index along the longest path through the graph 

774 :type graph_index: list[list[tuple[str, str]]] 

775 :param peptide_mismatches: list of peptides that did not match to the reference 

776 sequence 

777 :type peptide_mismatches: list[str] 

778 :param allowed_mismatches: number of mismatches allowed for a peptide to be 

779 considered a match 

780 :type allowed_mismatches: int 

781 :param graph: protein variation graph, as created by ProtGraph 

782 (-> _create_protein_variation_graph) 

783 :type graph: networkx.DiGraph 

784 :param longest_paths: length of longest path through the graph to each node 

785 :type longest_paths: dict[str, int] 

786 

787 :return: dict of peptide to dict of start index of peptide match to dict of node to 

788 tuple of start and end position of match in this node 

789 :rtype: dict[str, dict[int, dict[str, tuple[int, int]]]] 

790 """ 

791 

792 peptide_mismatches = set(peptide_mismatches) 

793 

794 def _match_on_graph( 

795 mismatches, 

796 allowed_mismatches, 

797 graph, 

798 current_node, 

799 left_over_peptide, 

800 node_match_data, 

801 current_index, 

802 ): 

803 """ 

804 Recursive function that matches a peptide to the graph. The function branches 

805 off at the end of each node that is matches to the end while there is still 

806 amino acids left over in the peptide. A new recursion is opened for each 

807 successor of a node, given the scenario above. A recursion is closed if the end 

808 of the peptide is reached or if the number of mismatches exceeds the allowed 

809 number of mismatches. 

810 For cases where multiple paths through the graph are possible to match to a 

811 given peptide from a starting position, the function will return the path with 

812 the least number of mismatches. If the number of mismatches is equal, 

813 the first of the equal options will be returned. 

814 

815 :param mismatches: number of mismatches so far 

816 :type mismatches: int 

817 :param allowed_mismatches: number of mismatches allowed per start position 

818 :type allowed_mismatches: int 

819 :param graph: protein variation graph, as created by ProtGraph 

820 (-> _create_protein_variation_graph) 

821 :type graph: networkx.DiGraph 

822 :param current_node: current node in the graph, starting with the node of the 

823 match start 

824 :type current_node: str 

825 :param left_over_peptide: peptide that still needs to be matched to the graph 

826 :type left_over_peptide: str 

827 :param node_match_data: dict of node to tuple of start position, end position 

828 :type node_match_data: dict 

829 :param current_index: index of the amino acid in the current node that is being 

830 matched to the peptide 

831 :type current_index: int 

832 

833 :return: tuple of bool, dict of node to tuple of start position, end position, 

834 number of mismatches 

835 :rtype: tuple[bool, dict[str, tuple[int, int]], int] 835 ↛ 836line 835 didn't jump to line 836, because the condition on line 835 was never true

836 """ 

837 

838 if mismatches > allowed_mismatches: 

839 return False, {}, mismatches 

840 if not left_over_peptide: 

841 return True, node_match_data, mismatches 

842 

843 # check if leftover peptide and rest of aminoacids in current node match 

844 # if so, add node and start and end position to node_match_data, 

845 # return True, node_match_data, mismatches 

846 last_index = current_index 

847 added_nodes = [] 

848 # enumerating because of index, would use zip_longest otherwise 

849 for i, label_aa in enumerate( 

850 graph.nodes[current_node]["aminoacid"][current_index:] 

851 ): 

852 if i > len(left_over_peptide) - 1: 

853 return True, node_match_data, mismatches 

854 if label_aa != left_over_peptide[i]: 

855 mismatches += 1 

856 if mismatches > allowed_mismatches: 

857 return False, {}, mismatches 

858 if current_node in added_nodes: 

859 node_match_data[current_node] = ( 

860 node_match_data[current_node][0], 

861 node_match_data[current_node][1] + 1, 

862 ) 

863 else: 

864 node_match_data[current_node] = (current_index, current_index) 

865 added_nodes.append(current_node) 

866 last_index = i 

867 

868 # node is matched til end, peptide not done 

869 data_from_successor = [] 

870 recursion_start_mismatches = mismatches 

871 for succ in graph.successors(current_node): 

872 recursion_start_data = node_match_data.copy() 

873 match, match_data_from_succ, mismatches = _match_on_graph( 

874 mismatches=recursion_start_mismatches, 

875 allowed_mismatches=allowed_mismatches, 

876 graph=graph, 

877 current_node=succ, 

878 left_over_peptide=left_over_peptide[last_index + 1 :], 

879 node_match_data=recursion_start_data, 

880 current_index=0, 

881 ) 

882 if match: 

883 data_from_successor.append((match, match_data_from_succ, mismatches)) 

884 if data_from_successor: 

885 return_val = min(data_from_successor, key=lambda item: item[2]) 

886 return return_val 

887 else: 

888 return False, {}, mismatches 

889 

890 peptide_match_info = {} 

891 for peptide, indices in potential_peptide_matches.items(): 

892 peptide_match_nodes = {} # store positions of matches for each node 

893 for match_start_index in indices: # start index is of ref_index 893 ↛ 898line 893 didn't jump to line 898, because the loop on line 893 didn't complete

894 # check with what node at the index the peptide start matches with 894 ↛ 893line 894 didn't jump to line 893, because the condition on line 894 was always true

895 starting_aa = peptide[0] 

896 for node, aa in graph_index[match_start_index]: 

897 if starting_aa == aa: 

898 starting_node = node 

899 break 

900 else: 

901 logger.error( 

902 f"No fitting node for match start position {match_start_index} " 

903 f"of {peptide} found" 

904 ) 

905 continue 

906 matched, node_match_data, mismatches = _match_on_graph( 

907 mismatches=0, 

908 allowed_mismatches=allowed_mismatches, 

909 graph=graph, 

910 current_node=starting_node, 

911 left_over_peptide=peptide, 

912 node_match_data={}, 

913 current_index=match_start_index - longest_paths[starting_node], 

914 ) 

915 if matched: 

916 logger.debug(f"matched {peptide} at {match_start_index}") 

917 logger.debug(f"match data: {node_match_data}") 

918 peptide_match_nodes[match_start_index] = node_match_data 

919 else: 

920 logger.debug(f"mismatched start pos {match_start_index} for {peptide}") 

921 if peptide_match_nodes: 

922 peptide_match_info[peptide] = peptide_match_nodes 

923 else: 

924 peptide_mismatches.add(peptide) 

925 

926 return peptide_match_info, list(peptide_mismatches)

929def _modify_graph(graph, contig_positions): 

930 """ 

931 Splits nodes of graph at into new contig nodes defined by the start- and 

932 end-positions in contig_positions. Adds `match` attribute to nodes, "true" when 

933 contig, "false" if not. Adds `peptide` attribute to match nodes with a string of ';' 

934 separated peptides that are responsible for the matchSink (__end__) current_node 

935 will end up without `match`-attribute. 

936 

937 :param graph: Protein Graph to be modified 

938 :type graph: nx.DiGraph 

939 :param contig_positions: Dict from current_node to contig-positions 

940 {current_node: [(start, end)]}. 

941 :type contig_positions: dict(list[tuple]) 

942 :param longest_paths: mapping from current_node to the longest path to current_node 

943 (-> _longest_paths()) 

944 :type longest_paths: dict 

945 

946 :return: modified protein graph, with contigs & not-matched AAs as nodes, indicated 

947 by current_node attribute `matched` 

948 """ 

949 

950 def _node_length(node): 

951 return len(graph.nodes[node]["aminoacid"]) 

952 

953 logger.info("updating graph to visualise peptide matches") 

954 for current_node, contigs in contig_positions.items(): 

955 chars_removed = 0 

956 for start, end, peptide in contigs: 

957 start = start - chars_removed 

958 end = end - chars_removed 

959 # all reset to clarify that they are used only in this loop 

960 first_node = None 

961 second_node = None 

962 third_node = None 

963 first_node_dict = None 

964 second_node_dict = None 

965 third_node_dict = None 

966 node_num = len(graph.nodes) 

967 

968 if start == 0 and end == _node_length(current_node) - 1: 

969 logger.debug( 

970 f"matched whole current_node {current_node}, contig: ({start, end})" 

971 ) 

972 nx.set_node_attributes( 

973 graph, 

974 {current_node: {"match": "true", "peptides": peptide}}, 

975 ) 

976 continue 

977 

978 # check if contig starts at beginning of current_node. 

979 # if not, create first_node 

980 if start != 0: 

981 first_node = f"n{node_num}" 

982 node_num += 1 

983 first_node_label = graph.nodes[current_node]["aminoacid"][:start] 

984 first_node_dict = dict( 

985 node_for_adding=first_node, 

986 aminoacid=first_node_label, 

987 match="false", 

988 ) 

989 

990 if end != _node_length(current_node) - 1 and first_node: 

991 # first_node, second_node and third_node 

992 second_node = f"n{node_num}" 

993 node_num += 1 

994 second_node_label = graph.nodes[current_node]["aminoacid"][ 

995 start : end + 1 

996 ] 

997 second_node_dict = dict( 

998 node_for_adding=second_node, 

999 aminoacid=second_node_label, 

1000 match="true", 

1001 peptides=peptide, 

1002 ) 

1003 

1004 # adopt current_node to be third_node 

1005 third_node_label = graph.nodes[current_node]["aminoacid"][end + 1 :] 

1006 third_node = current_node 

1007 third_node_dict = { 

1008 third_node: { 

1009 "aminoacid": third_node_label, 

1010 "match": "false", 

1011 } 

1012 } 

1013 

1014 elif end != _node_length(current_node) - 1 and not first_node: 

1015 # first_node and third_node, no second node 

1016 first_node = f"n{node_num}" 

1017 node_num += 1 

1018 first_node_label = graph.nodes[current_node]["aminoacid"][: end + 1] 

1019 first_node_dict = dict( 

1020 node_for_adding=first_node, 

1021 aminoacid=first_node_label, 

1022 match="true", 

1023 peptides=peptide, 

1024 ) 

1025 

1026 # adopt current_node to be third_node 

1027 third_node_label = graph.nodes[current_node]["aminoacid"][end + 1 :] 

1028 third_node = current_node 

1029 third_node_dict = { 

1030 third_node: { 

1031 "aminoacid": third_node_label, 

1032 "match": "false", 

1033 } 

1034 } 

1035 second_node = None 

1036 

1037 else: # before_node and match_node 

1038 # turn current_node into third_node 

1039 third_node_label = graph.nodes[current_node]["aminoacid"][start:] 

1040 third_node = current_node 

1041 third_node_dict = { 

1042 third_node: { 

1043 "aminoacid": third_node_label, 

1044 "match": "true", 

1045 "peptides": peptide, 

1046 } 

1047 } 

1048 second_node = None 

1049 

1050 # add edges 

1051 graph.add_node(**first_node_dict) 

1052 predecessors = list(graph.predecessors(current_node)) 

1053 for pre in predecessors: 

1054 graph.remove_edge(pre, current_node) 

1055 graph.add_edge(pre, first_node) 

1056 if second_node: 

1057 graph.add_node(**second_node_dict) 

1058 nx.set_node_attributes(graph, third_node_dict) 

1059 graph.add_edge(first_node, second_node) 

1060 graph.add_edge(second_node, third_node) 

1061 else: 

1062 nx.set_node_attributes(graph, third_node_dict) 

1063 graph.add_edge(first_node, third_node) 

1064 

1065 chars_removed += len(graph.nodes[first_node]["aminoacid"]) 

1066 if second_node: 

1067 chars_removed += len(graph.nodes[second_node]["aminoacid"]) 

1068 

1069 return graph

1072def _get_peptides( 

1073 peptide_df: pd.DataFrame, 

1074 protein_id: str, 

1075 grouping: str | None, 

1076 selected_groups: list | None, 

1077) -> list[str] | None: 

1078 """ 

1079 Get peptides for a protein ID from a peptide dataframe. 

1080 Includes all peptides where either just the protein ID occurs or it is in the group 

1081 of proteins associated with a given peptide. After selecting by protein id and if 

1082 the grouping and selected_groups are set, the peptides are filtered by the grouping 

1083 column, including everything that matches any of the selected groups. 

1084 

1085 :param peptide_df: Peptide dataframe 

1086 :type peptide_df: pd.DataFrame 

1087 :param protein_id: (UniProt) Protein ID 

1088 :type protein_id: str 

1089 :param grouping: Column name of the grouping column in the peptide dataframe 

1090 :type: str | None 

1091 :param selected_groups: List of groups from grouping column to include 

1092 :type: str | None 

1093 

1094 :return: List of peptides 

1095 :rtype: list[str] 

1096 """ 

1097 

1098 if grouping is not None: 

1099 assert ( 

1100 grouping in peptide_df.columns 

1101 ), f"Grouping '{grouping}' not found in peptide_df" 

1102 

1103 if selected_groups is not None: 

1104 if grouping is None: 

1105 raise ValueError("Grouping must be set if selected_groups is set") 

1106 for group in selected_groups: 

1107 assert ( 

1108 group in peptide_df[grouping].unique() 

1109 ), f"Group '{group}' not found in peptide_df column '{grouping}'" 

1110 

1111 df = peptide_df[peptide_df["Protein ID"].str.contains(protein_id)] 

1112 pattern = rf"^({protein_id}-\d+)$" 

1113 protein_id_filter = df["Protein ID"].str.contains(pattern) 

1114 df = df[~protein_id_filter] 

1115 

1116 intensity_name = [col for col in df.columns if "intensity" in col.lower()][0] 

1117 df = df.dropna(subset=[intensity_name]) 

1118 df = df[df[intensity_name] != 0] 

1119 

1120 # only get peptides where grouping-column matches at least one of selected_groups 

1121 if grouping is not None and selected_groups is not None: 

1122 mask = df[grouping].isin(selected_groups) 

1123 df = df[mask] 

1124 

1125 peptides = df["Sequence"].unique().tolist() 

1126 return peptides
